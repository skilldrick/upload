git init
#start a new git repo in current directory

git add .
#add all the files in the current directory to the staging area

git commit
#move everything from staging directory to repo

git commit -a
#commit everything without having to add

git push origin mybranch
#origin here is github, mybranch is the branch you want to push

git checkout mybranch
#change the current branch to mybranch

git tag mytag
#tag last commit with 'mytag'
